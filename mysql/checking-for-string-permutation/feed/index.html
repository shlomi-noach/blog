static bool checkPerm(string n1,string n2) {

if(n1.Length != n2.Length){
return false;
}

char[] res1 = n1.ToCharArray();
char[] res2 = n2.ToCharArray();
Array.Sort(res1); Array.Sort(res2);
string s1 = new string(res1);string s2 = new string(res2);

if(!s1.Equals(s2)){
return false;
}

return true;

}
